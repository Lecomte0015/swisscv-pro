    
    // GET /admin/users/:id/details - Détails complets d'un utilisateur
    if (request.method === "GET" && url.pathname.match(/^\/admin\/users\/[^\/]+\/details$/)) {
      try {
        const authHeader = request.headers.get("Authorization");
        if (!authHeader || !authHeader.startsWith("Bearer ")) {
          return new Response(JSON.stringify({
            success: false,
            error: "Authentication required"
          }), {
            status: 401,
            headers: corsHeaders
          });
        }
        
        const userId = url.pathname.split("/")[3];
        
        // Récupérer les infos utilisateur
        const user = await env.DB.prepare(`
          SELECT id, email, subscription_tier, subscription_status, created_at
          FROM users WHERE id = ?
        `).bind(userId).first();
        
        if (!user) {
          return new Response(JSON.stringify({
            success: false,
            error: "User not found"
          }), {
            status: 404,
            headers: corsHeaders
          });
        }
        
        // Récupérer les statistiques
        const cvAnalysesCount = await env.DB.prepare(`
          SELECT COUNT(*) as count FROM cv_analyses WHERE user_id = ?
        `).bind(userId).first();
        
        const jobApplicationsCount = await env.DB.prepare(`
          SELECT COUNT(*) as count FROM job_applications WHERE user_id = ?
        `).bind(userId).first();
        
        // Récupérer les dernières analyses CV
        const recentAnalyses = await env.DB.prepare(`
          SELECT id, score, ats_score, created_at
          FROM cv_analyses
          WHERE user_id = ?
          ORDER BY created_at DESC
          LIMIT 5
        `).bind(userId).all();
        
        // Récupérer les dernières candidatures
        const recentApplications = await env.DB.prepare(`
          SELECT id, job_title, company, status, applied_date
          FROM job_applications
          WHERE user_id = ?
          ORDER BY applied_date DESC
          LIMIT 5
        `).bind(userId).all();
        
        // Calculer jours depuis inscription
        const daysSinceSignup = Math.floor((Date.now() / 1000 - user.created_at) / 86400);
        
        return new Response(JSON.stringify({
          success: true,
          user: {
            id: user.id,
            email: user.email,
            tier: user.subscription_tier || 'free',
            subscription_status: user.subscription_status,
            created_at: user.created_at
          },
          stats: {
            total_analyses: cvAnalysesCount?.count || 0,
            total_job_applications: jobApplicationsCount?.count || 0,
            days_since_signup: daysSinceSignup
          },
          cv_analyses: recentAnalyses.results || [],
          recent_applications: recentApplications.results || []
        }), {
          status: 200,
          headers: corsHeaders
        });
        
      } catch (error) {
        console.error("Error fetching user details:", error);
        return new Response(JSON.stringify({
          success: false,
          error: "Internal server error",
          details: error.message
        }), {
          status: 500,
          headers: corsHeaders
        });
      }
    }
