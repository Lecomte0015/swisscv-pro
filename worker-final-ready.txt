async function createSystemNotification(env, userId, options) {
  const { type, tier, limit, remaining } = options;
  let title = '', message = '', link = null;
  
  if (type === 'welcome') {
    title = tier === 'free' ? 'Bienvenue sur SwissCV Pro !' : `Bienvenue sur SwissCV Pro ${tier.toUpperCase()} !`;
    message = tier === 'free' ? 'Commencez par analyser votre CV.' : `Profitez des fonctionnalités ${tier.toUpperCase()}.`;
    link = '/app/app.html';
  } else if (type === 'limit_reached') {
    title = 'Limite gratuite atteinte';
    message = `Vous avez utilisé vos ${limit} analyses gratuites. Passez à Premium !`;
    link = '/pricing.html';
  } else if (type === 'limit_warning') {
    title = 'Dernière analyse gratuite';
    message = `Il vous reste ${remaining} analyse. Passez à Premium !`;
    link = '/pricing.html';
  }
  
  return await createNotification(env, userId, type, title, message, link);
}
