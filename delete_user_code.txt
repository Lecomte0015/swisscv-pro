        const userId = url.pathname.split("/")[3];
        
        // Supprimer toutes les données liées à l'utilisateur
        // (pour éviter les erreurs de contrainte de clé étrangère)
        await env.DB.prepare(`DELETE FROM user_profiles WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM experiences WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM user_skills WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM user_languages WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM certifications WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM career_goals WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM cv_analyses WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM job_applications WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM free_usage WHERE user_id = ?`).bind(userId).run();
        await env.DB.prepare(`DELETE FROM job_matches WHERE user_id = ?`).bind(userId).run();
        
        // Enfin, supprimer l'utilisateur
        await env.DB.prepare(`DELETE FROM users WHERE id = ?`).bind(userId).run();
        
        return new Response(JSON.stringify({
          success: true,
          message: "User deleted successfully"
        }), {
          status: 200,
          headers: corsHeaders
        });
